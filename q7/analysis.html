Python Code Used to Generate Analysis

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import base64
from io import BytesIO

# load dataset
df = pd.read_csv("employees.csv")

# frequency of Marketing department
marketing_count = df[df["department"] == "Marketing"].shape[0]
print("Marketing department count:", marketing_count)

# histogram
plt.figure(figsize=(8, 6))
sns.histplot(df["department"], discrete=True)
plt.title("Employee Count by Department")
plt.xlabel("Department")
plt.ylabel("Count")

# convert to Base64
buf = BytesIO()
plt.savefig(buf, format="png", bbox_inches="tight")
buf.seek(0)
img_base64 = base64.b64encode(buf.read()).decode("utf-8")
print(img_base64)
