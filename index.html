import pandas as pd

# Quarterly revenue and expenses (simplified)
df = pd.DataFrame({
    "quarter": ["Q1", "Q2"],
    "revenue": [1_100_000_000, 1_250_000_000],
    "expense": [900_000_000, 1_040_000_000],
})

df["net_income"] = df["revenue"] - df["expense"]
df["net_margin"] = df["net_income"] / df["revenue"]

# EPS calculation (simplified)
shares_outstanding = 145_000_000
df["eps"] = df["net_income"] / shares_outstanding

print(df)

Key Financial Formulas

Net Income:

\[ \text{Net Income} = \text{Revenue} - \text{Expenses} \]

Earnings Per Share (EPS):

\[ \text{EPS} = \frac{\text{Net Income}}{\text{Shares Outstanding}} \]

Return on Equity (ROE):

\[ \text{ROE} = \frac{\text{Net Income}}{\text{Average Shareholders' Equity}} \]
